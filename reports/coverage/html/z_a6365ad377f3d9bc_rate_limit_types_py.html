94@dataclass 

95class TokenBucket: 

96 """Token bucket rate limiter implementation.""" 

97 capacity: float 

98 tokens: float 

99 refill_rate: float 

100 last_refill: float = field(default_factory=time.time) 

101 

102 def consume(self, tokens: float = 1.0) -> bool: 

103 """Consume tokens from bucket.""" 

104 self._refill() 

105 if self.tokens >= tokens: 

106 self.tokens -= tokens 

107 return True 

108 return False 

109 

110 def _refill(self) -> None: 

111 """Refill tokens based on elapsed time.""" 

112 now = time.time() 

113 elapsed = now - self.last_refill 

114 self.tokens = min(self.capacity, self.tokens + elapsed * self.refill_rate) 

115 self.last_refill = now 

116 

117 def get_available_tokens(self) -> float: 

118 """Get number of available tokens.""" 

119 self._refill() 

120 return self.tokens 

121 

122 def reset(self) -> None: 

123 """Reset bucket to full capacity.""" 

124 self.tokens = self.capacity 

125 self.last_refill = time.time()

182class SlidingWindowCounter: 

183 """Sliding window counter for rate limiting.""" 

184 

185 def __init__(self, window_size: int, bucket_size: int = 60): 

186 self.window_size = window_size 

187 self.bucket_size = bucket_size 

188 self.buckets: Dict[int, int] = defaultdict(int) 

189 

190 def add_request(self, timestamp: float) -> None: 

191 """Add a request at given timestamp.""" 

192 bucket = int(timestamp // self.bucket_size) 

193 self.buckets[bucket] += 1 

194 

195 def get_count(self, timestamp: float) -> int: 

196 """Get request count in sliding window.""" 

197 current_bucket = int(timestamp // self.bucket_size) 

198 cutoff = self._calculate_cutoff_bucket(current_bucket) 

199 return self._sum_valid_buckets(cutoff) 

200 

201 def _calculate_cutoff_bucket(self, current_bucket: int) -> int: 

202 """Calculate cutoff bucket for sliding window.""" 

203 return current_bucket - (self.window_size // self.bucket_size) 

204 

205 def _sum_valid_buckets(self, cutoff: int) -> int: 

206 """Sum requests from valid buckets within window.""" 

207 return sum(requests for bucket, requests in self.buckets.items() if bucket > cutoff) 

208 

209 def cleanup(self, timestamp: float) -> None: 

210 """Clean up old buckets.""" 

211 current_bucket = int(timestamp // self.bucket_size) 

212 cutoff = current_bucket - (self.window_size // self.bucket_size) - 1 

213 

214 expired = [b for b in self.buckets.keys() if b < cutoff] 

215 for bucket in expired: 

216 del self.buckets[bucket]
