214 async def _discover_tools(self, categories: List[str], entities: ExtractedEntities) -> List[ToolRecommendation]: 

215 """Discover tools for the identified categories.""" 

216 all_recommendations = [] 

217 

218 for category in categories: 

219 recommendations = self.tool_recommender.recommend_tools(category, entities) 

220 all_recommendations.extend(recommendations) 

221 

222 # Remove duplicates and sort by relevance 

223 unique_tools = {} 

224 for rec in all_recommendations: 

225 if rec.tool_name not in unique_tools or rec.relevance_score > unique_tools[rec.tool_name].relevance_score: 

226 unique_tools[rec.tool_name] = rec 

227 

228 return sorted(unique_tools.values(), key=lambda x: x.relevance_score, reverse=True)[:10]
