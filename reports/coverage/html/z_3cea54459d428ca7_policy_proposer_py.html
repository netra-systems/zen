25 def _get_member_spans(self, pattern: DiscoveredPattern, span_map: Dict[str, UnifiedLogEntry]) -> List[UnifiedLogEntry]: 

26 """Get member spans for a pattern""" 

27 return [span_map[sid] for sid in pattern.member_span_ids if sid in span_map]

29 async def _simulate_all_outcomes(self, context: ToolContext, pattern: DiscoveredPattern, 

30 all_options: List, user_goal: str, representative_span: UnifiedLogEntry) -> List: 

31 """Simulate outcomes for all supply options""" 

32 sim_tasks = [self._simulate_policy_outcome(context, pattern, supply, user_goal, representative_span) for supply in all_options] 

33 policy_outcomes = [o for o in await asyncio.gather(*sim_tasks) if o] 

34 return sorted(policy_outcomes, key=lambda x: x.utility_score, reverse=True)

36 def _calculate_baseline_metrics(self, member_spans: List[UnifiedLogEntry]) -> Dict[str, float]: 

37 """Calculate baseline metrics for member spans""" 

38 return { 

39 "avg_cost_usd": sum(s.finops.total_cost_usd for s in member_spans) / len(member_spans), 

40 "avg_latency_ms": sum(s.performance.latency_ms.total_e2e_ms for s in member_spans) / len(member_spans), 

41 "avg_quality_score": sum(s.quality.score for s in member_spans) if all(s.quality for s in member_spans) else 0.8, 

42 }

44 def _calculate_pattern_impact(self, member_spans: List[UnifiedLogEntry], span_map: Dict[str, UnifiedLogEntry]) -> float: 

45 """Calculate pattern impact fraction""" 

46 pattern_spend = sum(s.finops.total_cost_usd for s in member_spans) 

47 all_spans_spend = sum(s.finops.total_cost_usd for s in span_map.values()) 

48 return (pattern_spend / all_spans_spend) if all_spans_spend > 0 else 0

50 def _create_learned_policy(self, pattern: DiscoveredPattern, sorted_outcomes: List, 

51 baseline_metrics: Dict, pattern_impact_fraction: float) -> LearnedPolicy: 

52 """Create a learned policy from outcomes""" 

53 return LearnedPolicy( 

54 pattern_name=pattern.pattern_name, optimal_supply_option_name=sorted_outcomes[0].supply_option_name, 

55 predicted_outcome=sorted_outcomes[0], alternative_outcomes=sorted_outcomes[1:4], 

56 baseline_metrics=baseline_metrics, pattern_impact_fraction=pattern_impact_fraction)

58 async def run( 

59 self, context: ToolContext, patterns: List[DiscoveredPattern], span_map: Dict[str, UnifiedLogEntry] 

60 ) -> Tuple[List[LearnedPolicy], List[PredictedOutcome]]: 

61 """Finds the best routing policies through simulation.""" 

62 policies, outcomes, all_options = [], [], await self._get_supply_catalog(context) 

63 for pattern in patterns: 

64 member_spans = self._get_member_spans(pattern, span_map) 

65 if not member_spans: 

66 continue 

67 sorted_outcomes = await self._simulate_all_outcomes(context, pattern, all_options, member_spans[0].request.user_goal, member_spans[0]) 

68 if not sorted_outcomes: 

69 continue 

70 outcomes.extend(sorted_outcomes) 

71 baseline_metrics = self._calculate_baseline_metrics(member_spans) 

72 pattern_impact_fraction = self._calculate_pattern_impact(member_spans, span_map) 

73 policies.append(self._create_learned_policy(pattern, sorted_outcomes, baseline_metrics, pattern_impact_fraction)) 

74 return policies, outcomes

102 async def _execute_llm_simulation(self, context: ToolContext, prompt: str) -> Any: 

103 """Execute LLM simulation call""" 

104 llm = context.llm_manager.get_llm(self.llm_name or "default")

116 async def _simulate_policy_outcome(self, context: ToolContext, pattern: DiscoveredPattern, supply_option: SupplyOption, user_goal: str, span: UnifiedLogEntry) -> PredictedOutcome:
