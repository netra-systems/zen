216 def _format_period_description(self, start: datetime, end: datetime) -> str: 

217 """Format billing period description.""" 

218 if start.month == end.month and start.year == end.year: 

219 return f"{start.strftime('%B %Y')}" 

220 else: 

221 return f"{start.strftime('%B %d, %Y')} - {end.strftime('%B %d, %Y')}"
