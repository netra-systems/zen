40 def get_cache_key(self, token: str, token_type: str = "access") -> str: 

41 """Generate consistent cache key for token validation""" 

42 token_hash = hashlib.sha256(token.encode()).hexdigest()[:16] 

43 return f"jwt_validation:{token_hash}:{token_type}"

45 def get_from_cache(self, cache_key: str) -> Optional[Any]: 

46 """Get validation result from cache with Redis fallback""" 

47 try: 

48 self._validation_stats['validation_count'] += 1 

49 

50 # Try in-memory cache first (fastest - sub-millisecond) 

51 if cache_key in self._validation_cache: 

52 cached_entry = self._validation_cache[cache_key] 

53 if cached_entry['expires'] > time.time(): 

54 # Also check if the cached JWT token itself has expired 

55 cached_data = cached_entry['data'] 

56 if cached_data and isinstance(cached_data, dict): 

57 jwt_exp = cached_data.get('exp') 

58 if jwt_exp and jwt_exp <= time.time(): 

59 # JWT token has expired, remove from cache 

60 del self._validation_cache[cache_key] 

61 return None 

62 self._validation_stats['cache_hits'] += 1 

63 return cached_data 

64 else: 

65 # Expired, remove from memory cache 

66 del self._validation_cache[cache_key] 

67 

68 # Try Redis cache if available (fast - 1-5ms typically) 

69 if self._cache_enabled and self.redis_manager.get_client(): 

70 try: 

71 redis_data = self.redis_manager.get_client().get(f"jwt_cache:{cache_key}") 

72 if redis_data: 

73 cached_data = json.loads(redis_data) 

74 

75 # Check if the cached JWT token itself has expired 

76 if cached_data and isinstance(cached_data, dict): 

77 jwt_exp = cached_data.get('exp') 

78 if jwt_exp and jwt_exp <= time.time(): 

79 # JWT token has expired, remove from Redis cache 

80 self.redis_manager.get_client().delete(f"jwt_cache:{cache_key}") 

81 return None 

82 

83 self._validation_stats['redis_hits'] += 1 

84 

85 # Also update in-memory cache for next time 

86 self._validation_cache[cache_key] = { 

87 'data': cached_data, 

88 'expires': time.time() + min(300, 60) # 1-5 min memory cache 

89 } 

90 return cached_data 

91 else: 

92 self._validation_stats['redis_misses'] += 1 

93 except Exception as e: 

94 logger.debug(f"Redis cache read failed: {e}") 

95 self._validation_stats['redis_misses'] += 1 

96 

97 self._validation_stats['cache_misses'] += 1 

98 return None 

99 

100 except Exception as e: 

101 logger.debug(f"Cache read error: {e}") 

102 return None

104 def cache_validation_result(self, cache_key: str, result: Optional[Dict], ttl: int = None): 

105 """Cache validation result with Redis persistence and memory cache""" 

106 try: 

107 if ttl is None: 

108 ttl = self._cache_ttl 

109 

110 cache_data = result if result is not None else "INVALID" 

111 

112 # Update in-memory cache (with size limit) 

113 if len(self._validation_cache) >= self._max_cache_size: 

114 # Remove oldest entries 

115 current_time = time.time() 

116 expired_keys = [k for k, v in self._validation_cache.items() 

117 if v['expires'] <= current_time] 

118 for key in expired_keys[:len(expired_keys)//2]: # Remove half of expired 

119 del self._validation_cache[key] 

120 

121 # Cache in memory for fastest access 

122 self._validation_cache[cache_key] = { 

123 'data': cache_data, 

124 'expires': time.time() + min(ttl, 300) # Max 5 min memory cache 

125 } 

126 

127 # Update Redis cache if available (async to not block) 

128 if self._cache_enabled and self.redis_manager.get_client(): 

129 try: 

130 asyncio.create_task(self._cache_to_redis_async(cache_key, cache_data, ttl)) 

131 except Exception as e: 

132 logger.debug(f"Redis cache write failed: {e}") 

133 

134 except Exception as e: 

135 logger.debug(f"Cache write error: {e}")

137 async def _cache_to_redis_async(self, cache_key: str, cache_data: Any, ttl: int): 

138 """Async Redis cache write to avoid blocking""" 

139 try: 

140 redis_client = self.redis_manager.get_client() 

141 if redis_client: 

142 redis_client.setex( 

143 f"jwt_cache:{cache_key}", 

144 ttl, 

145 json.dumps(cache_data, default=str) 

146 ) 

147 except Exception as e: 

148 logger.debug(f"Async Redis cache write failed: {e}")

150 def invalidate_user_cache(self, user_id: str): 

151 """Invalidate all cached tokens for a user (when user is blacklisted)""" 

152 try: 

153 # Remove from memory cache 

154 keys_to_remove = [] 

155 for cache_key, cached_entry in self._validation_cache.items(): 

156 if (isinstance(cached_entry['data'], dict) and 

157 cached_entry['data'].get('sub') == user_id): 

158 keys_to_remove.append(cache_key) 

159 

160 for key in keys_to_remove: 

161 del self._validation_cache[key] 

162 

163 # Remove from Redis cache if available 

164 if self._cache_enabled and self.redis_manager.get_client(): 

165 try: 

166 # This is a simplified approach - in production you might want 

167 # to maintain an index of user_id -> cache_keys 

168 pattern = f"jwt_cache:*" 

169 keys = self.redis_manager.get_client().keys(pattern) 

170 for key in keys: 

171 try: 

172 data = self.redis_manager.get_client().get(key) 

173 if data: 

174 token_data = json.loads(data) 

175 if (isinstance(token_data, dict) and 

176 token_data.get('sub') == user_id): 

177 self.redis_manager.get_client().delete(key) 

178 except: 

179 continue 

180 

181 except Exception as e: 

182 logger.debug(f"Redis user cache invalidation failed: {e}") 

183 

184 logger.info(f"Invalidated cached tokens for user {user_id}") 

185 

186 except Exception as e: 

187 logger.error(f"User cache invalidation error: {e}")

202 def get_cache_stats(self) -> Dict[str, Any]: 

203 """Get cache performance statistics""" 

204 total_validations = self._validation_stats['validation_count'] 

205 memory_hit_rate = (self._validation_stats['cache_hits'] / total_validations * 100) if total_validations > 0 else 0 

206 redis_hit_rate = (self._validation_stats['redis_hits'] / total_validations * 100) if total_validations > 0 else 0 

207 overall_hit_rate = ((self._validation_stats['cache_hits'] + self._validation_stats['redis_hits']) / total_validations * 100) if total_validations > 0 else 0 

208 

209 return { 

210 'validation_count': total_validations, 

211 'memory_cache_hits': self._validation_stats['cache_hits'], 

212 'redis_cache_hits': self._validation_stats['redis_hits'], 

213 'cache_misses': self._validation_stats['cache_misses'], 

214 'memory_hit_rate_percent': round(memory_hit_rate, 2), 

215 'redis_hit_rate_percent': round(redis_hit_rate, 2), 

216 'overall_hit_rate_percent': round(overall_hit_rate, 2), 

217 'memory_cache_size': len(self._validation_cache), 

218 'cache_enabled': self._cache_enabled 

219 }
