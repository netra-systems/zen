516 def _can_execute_agent(self, agent_name: str, completed_agents: Set[str], context_metadata: Dict[str, Any]) -> tuple[bool, List[str]]: 

517 """Check if an agent can be executed based on its dependencies. 

518  

519 This is the SSOT for dependency validation. Checks both agent completion 

520 and metadata availability. 

521  

522 Args: 

523 agent_name: Name of the agent to check 

524 completed_agents: Set of successfully completed agents 

525 context_metadata: Current context metadata with results 

526  

527 Returns: 

528 Tuple of (can_execute, missing_dependencies) 

529 """ 

530 dep_spec = self.AGENT_DEPENDENCIES.get(agent_name, {}) 

531 required_deps = dep_spec.get("required", []) 

532 missing_deps = [] 

533 

534 for dep_agent, metadata_key in required_deps: 

535 # Check both: agent completion AND metadata availability 

536 if dep_agent not in completed_agents: 

537 missing_deps.append(f"{dep_agent} (not executed)") 

538 logger.warning(f"Agent {agent_name} missing dependency: {dep_agent} not executed") 

539 elif metadata_key not in context_metadata: 

540 missing_deps.append(f"{dep_agent} (no {metadata_key})") 

541 logger.warning(f"Agent {agent_name} missing dependency: {metadata_key} not in context") 

542 

543 # Log optional dependencies for debugging 

544 optional_deps = dep_spec.get("optional", []) 

545 for dep_agent, metadata_key in optional_deps: 

546 if dep_agent not in completed_agents: 

547 logger.debug(f"Agent {agent_name} optional dependency missing: {dep_agent}") 

548 elif metadata_key not in context_metadata: 

549 logger.debug(f"Agent {agent_name} optional metadata missing: {metadata_key}") 

550 

551 can_execute = len(missing_deps) == 0 

552 return can_execute, missing_deps

586 logger.info("📊 Executing triage agent to determine optimal workflow...") 

587 triage_result = await self._execute_single_agent( 

588 agent_instances["triage"], context, "triage", completed_agents, failed_agents 

589 ) 

590 results["triage"] = triage_result 

591 logger.info(f"✅ Triage completed. Data sufficiency: {triage_result.get('data_sufficiency', 'unknown')}") 

592 except Exception as e: 

593 logger.warning(f"⚠️ Triage failed (non-critical): {e}") 

594 logger.info("📌 Continuing with UVS fallback workflow...") 

595 results["triage"] = { 

596 "error": str(e), 

597 "status": "failed", 

598 "fallback": "Using default workflow without triage insights" 

599 } 

600 failed_agents.add("triage") 

601 

602 # Step 2: Determine dynamic execution order based on triage 

603 execution_order = self._determine_execution_order(triage_result, context) 

604 logger.info(f"Dynamic execution order based on triage: {execution_order}") 

605 

606 # Step 3: Execute determined agents 

607 for agent_name in execution_order: 

608 # Skip if agent not available 

609 if agent_name not in agent_instances: 

610 logger.debug(f"Agent {agent_name} not in available instances, skipping") 

611 continue 

612 

613 # Check dependencies before execution (SSOT validation) 

614 can_execute, missing_deps = self._can_execute_agent(agent_name, completed_agents, context.metadata) 

615 if not can_execute: 

616 # UVS: Dependencies not met is not a failure - just skip 

617 logger.info(f"⏭️ Skipping {agent_name}: optional dependencies not met: {missing_deps}") 

618 results[agent_name] = { 

619 "status": "skipped", 

620 "reason": "Optional dependencies not available", 

621 "missing_deps": missing_deps 

622 } 

623 # Don't add to failed_agents - this is expected behavior 

624 continue 

625 

626 try: 

627 start_time = time.time() 

628 await self._emit_thinking(context, f"Running {agent_name} with isolated instance...") 

629 

630 # Create child context for sub-agent 

631 child_context = context.create_child_context( 

632 operation_name=f"{agent_name}_execution", 

633 additional_metadata={ 

634 "agent_name": agent_name, 

635 "flow_id": flow_id 

636 } 

637 ) 

638 

639 # Execute agent with child context and retry logic 

640 agent_result = await self._execute_agent_with_retry( 

641 agent_instances[agent_name], child_context, agent_name 

642 ) 

643 

644 execution_time = time.time() - start_time 

645 logger.info(f"⏱️ Agent {agent_name} executed in {execution_time:.2f}s for user {context.user_id}") 

646 

647 # Validate result is not an error 

648 if isinstance(agent_result, dict) and agent_result.get("status") == "failed": 

649 raise RuntimeError(f"Agent returned failure status: {agent_result.get('error', 'Unknown error')}") 

650 

651 results[agent_name] = agent_result 

652 completed_agents.add(agent_name) 

653 

654 # CRITICAL: Propagate metadata from child context back to parent context 

655 # This ensures results like triage_result, data_result, etc. are available for reporting 

656 self._merge_child_metadata_to_parent(context, child_context, agent_name) 

657 

658 # Store agent result under expected keys (SSOT) 

659 self._store_agent_result(context, agent_name, agent_result) 

660 

661 logger.info(f"✅ Agent {agent_name} completed successfully for user {context.user_id}") 

662 

663 # Add small delay between agents to prevent overwhelming the system 

664 # Only if execution was very fast (< 0.5s) 

665 if execution_time < 0.5: 

666 delay = 0.5 - execution_time 

667 logger.debug(f"Adding {delay:.2f}s delay after fast execution") 

668 await asyncio.sleep(delay) 

669 

670 except Exception as e: 

671 # Check if this agent can fail gracefully (UVS principle) 

672 agent_config = self.AGENT_DEPENDENCIES.get(agent_name, {}) 

673 can_fail = agent_config.get("can_fail", True) 

674 

675 if can_fail: 

676 # Non-critical agent - log and continue 

677 logger.warning(f"⚠️ Optional agent {agent_name} failed (non-critical): {e}") 

678 results[agent_name] = { 

679 "error": str(e), 

680 "status": "failed", 

681 "recoverable": True, 

682 "impact": "Minimal - other agents will compensate" 

683 } 

684 failed_agents.add(agent_name) 

685 

686 # Emit user-friendly message 

687 await self._emit_thinking(context, 

688 f"Note: {agent_name} encountered an issue but we're continuing with alternative approaches...") 

689 else: 

690 # Critical agent (only reporting) - must handle specially 

691 logger.error(f"❌ CRITICAL: Required agent {agent_name} failed: {e}", exc_info=True) 

692 

693 if agent_name == "reporting": 

694 # Reporting MUST succeed - attempt fallback 

695 logger.info("🔄 Attempting fallback reporting...") 

696 try: 

697 results["reporting"] = await self._create_fallback_report(context, results) 

698 logger.info("✅ Fallback reporting succeeded") 

699 completed_agents.add("reporting") 

700 except Exception as fallback_error: 

701 logger.error(f"❌ Fallback reporting also failed: {fallback_error}") 

702 # Last resort - create minimal report 

703 results["reporting"] = { 

704 "status": "emergency_fallback", 

705 "message": "System encountered an issue. Please try again.", 

706 "error": str(e) 

707 } 

708 else: 

709 results[agent_name] = {"error": str(e), "status": "failed"} 

710 failed_agents.add(agent_name) 

711 

712 # Log workflow summary 

713 logger.info(f"Workflow completed. Successful: {completed_agents}, Failed: {failed_agents}") 

714 

715 # Add workflow metadata to results 

716 results["_workflow_metadata"] = { 

717 "completed_agents": list(completed_agents), 

718 "failed_agents": list(failed_agents), 

719 "total_agents": len(execution_order), 

720 "success_rate": len(completed_agents) / len(execution_order) if execution_order else 0 

721 } 

722 

723 return results

725 async def _execute_agent_with_retry(self, 

726 agent: BaseAgent, 

727 context: UserExecutionContext, 

728 agent_name: str) -> Any: 

729 """Execute agent with exponential backoff retry logic. 

730  

731 Args: 

732 agent: Agent instance to execute 

733 context: User execution context (child context) 

734 agent_name: Name of agent for logging 

735  

736 Returns: 

737 Agent execution result 

738  

739 Raises: 

740 RuntimeError: If all retries are exhausted 

741 """ 

742 max_retries = 3 

743 base_delay = 1.0 # Start with 1 second 

744 max_delay = 16.0 # Cap at 16 seconds 

745 

746 for attempt in range(max_retries): 

747 try: 

748 if attempt > 0: 

749 # Calculate exponential backoff delay 

750 delay = min(base_delay * (2 ** (attempt - 1)), max_delay) 

751 logger.info(f"Retrying {agent_name} after {delay:.1f}s delay (attempt {attempt + 1}/{max_retries})") 

752 await asyncio.sleep(delay) 

753 await self._emit_thinking(context, f"Retrying {agent_name} (attempt {attempt + 1}/{max_retries})...") 

754 

755 # Execute the agent 

756 result = await self._execute_agent_with_context(agent, context, agent_name) 

757 

758 # Success - return result 

759 if attempt > 0: 

760 logger.info(f"✅ {agent_name} succeeded after {attempt + 1} attempts") 

761 return result 

762 

763 except Exception as e: 

764 logger.warning(f"Agent {agent_name} attempt {attempt + 1} failed: {e}") 

765 

766 # Check if this is a recoverable error 

767 if not self._is_recoverable_error(e): 

768 logger.error(f"Non-recoverable error for {agent_name}: {e}") 

769 raise 

770 

771 # Last attempt - raise the error 

772 if attempt == max_retries - 1: 

773 logger.error(f"Agent {agent_name} failed after {max_retries} attempts") 

774 raise RuntimeError(f"Agent {agent_name} failed after {max_retries} retries: {e}") from e

906 def _determine_execution_order(self, triage_result: Optional[Dict], context: UserExecutionContext) -> List[str]: 

907 """Determine dynamic execution order based on triage results. 

908  

909 UVS SIMPLIFIED FLOW: 

910 - Reporting ALWAYS runs (it's the only truly required agent) 

911 - Other agents run based on data availability and user intent 

912 - Skip unnecessary agents for faster responses 

913  

914 Args: 

915 triage_result: Result from triage agent (may be None or failed) 

916 context: User execution context 

917  

918 Returns: 

919 List of agent names to execute in order 

920 """ 

921 execution_order = [] 

922 

923 # Check if triage provided guidance 

924 if triage_result and isinstance(triage_result, dict): 

925 # Extract workflow hints from triage 

926 data_sufficiency = triage_result.get("data_sufficiency", "unknown") 

927 user_intent = triage_result.get("intent", {}) 

928 next_agents = triage_result.get("next_agents", []) 

929 

930 # Use triage's recommended workflow if available 

931 if next_agents: 

932 logger.info(f"Using triage-recommended workflow: {next_agents}") 

933 execution_order = [agent for agent in next_agents if agent != "reporting"] 

934 else: 

935 # Determine workflow based on data sufficiency 

936 if data_sufficiency == "sufficient": 

937 # Have data - selective pipeline based on intent 

938 execution_order = [] 

939 

940 # Only add agents that are actually needed 

941 if self._needs_data_analysis(user_intent, context): 

942 execution_order.append("data") 

943 

944 if self._needs_optimization(user_intent, context): 

945 execution_order.append("optimization") 

946 

947 if self._needs_action_plan(user_intent, context): 

948 execution_order.append("actions") 

949 

950 # If no specific agents needed, use data helper for guidance 

951 if not execution_order: 

952 execution_order = ["data_helper"] 

953 

954 elif data_sufficiency == "partial": 

955 # Some data - augment with helper 

956 execution_order = ["data_helper"] 

957 

958 # Add other agents if beneficial 

959 if self._needs_data_analysis(user_intent, context): 

960 execution_order.append("data") 

961 

962 if self._needs_optimization(user_intent, context): 

963 execution_order.append("optimization") 

964 

965 else: 

966 # No data or unknown - guidance flow 

967 execution_order = ["data_helper"] 

968 else: 

969 # Triage failed or no result - minimal flow 

970 logger.info("Triage unavailable. Using minimal UVS flow: Data Helper → Reporting") 

971 execution_order = ["data_helper"] 

972 

973 # CRITICAL: ALWAYS end with Reporting (UVS - can handle ANY scenario) 

974 execution_order.append("reporting") 

975 

976 # Log the dynamic workflow for debugging 

977 logger.info(f"UVS Dynamic workflow determined: {' → '.join(execution_order)}") 

978 logger.info(f"Workflow reasoning: data_sufficiency={triage_result.get('data_sufficiency', 'unknown') if triage_result else 'no_triage'}") 

979 

980 return execution_order

1176 def _is_recoverable_error(self, error: Exception) -> bool: 

1177 """Determine if an error is recoverable and worth retrying. 

1178  

1179 Args: 

1180 error: The exception to check 

1181  

1182 Returns: 

1183 True if the error is recoverable 

1184 """ 

1185 # Network and timeout errors are recoverable 

1186 recoverable_errors = [ 

1187 "timeout", "timed out", "connection", "network", 

1188 "temporary", "unavailable", "rate limit", "throttl" 

1189 ] 

1190 

1191 error_str = str(error).lower() 

1192 

1193 # Check for known recoverable patterns 

1194 for pattern in recoverable_errors: 

1195 if pattern in error_str: 

1196 return True 

1197 

1198 # Check for specific exception types 

1199 import asyncio 

1200 if isinstance(error, (asyncio.TimeoutError, ConnectionError, TimeoutError)): 

1201 return True 

1202 

1203 # Non-recoverable errors 

1204 non_recoverable = [ 

1205 "authentication", "permission", "forbidden", "unauthorized", 

1206 "invalid", "not found", "does not exist", "migration required" 

1207 ] 

1208 

1209 for pattern in non_recoverable: 

1210 if pattern in error_str: 

1211 return False
