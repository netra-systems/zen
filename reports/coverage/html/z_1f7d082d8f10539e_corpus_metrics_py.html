91 async def track_operation(self, corpus_id: str, operation_type: str, context: Optional[Dict[str, Any]] = None): 

92 """Context manager for tracking corpus operations""" 

93 operation_id = await self.core_collector.start_operation(corpus_id, operation_type) 

94 await self._take_operation_snapshot(operation_id, corpus_id) 

95 try: 

96 yield operation_id 

97 await self._complete_operation_tracking(operation_id, True, context) 

98 except Exception as e: 

99 await self._complete_operation_tracking(operation_id, False, {"error": str(e)}) 

100 raise

107 async def _complete_operation_tracking(self, operation_id: str, success: bool, context: Optional[Dict[str, Any]] = None) -> None: 

108 """Complete operation tracking with all collectors""" 

109 records_processed = context.get("records_processed") if context else None 

110 error_message = context.get("error") if context and not success else None 

111 operation_metrics = await self.core_collector.end_operation(operation_id, success, records_processed, error_message) 

112 await self._process_operation_completion(operation_id, operation_metrics)

115 """Process completed operation metrics""" 

116 if not (self.enable_resource_monitoring and operation_metrics):

118 resource_usage = await self.resource_monitor.calculate_operation_usage(operation_id) 

119 if resource_usage and self.enable_real_time_updates:

204 async def _assess_corpus_health(self, corpus_id: str, quality_metrics: Optional[QualityMetrics]) -> str: 

205 """Assess overall corpus health status""" 

206 success_rate = self.core_collector.get_success_rate(corpus_id) 

207 quality_score = quality_metrics.overall_score if quality_metrics else 0.5 

208 resource_alerts = self.resource_monitor.get_resource_alerts() 

209 critical_alerts = [a for a in resource_alerts if a.get("severity") == "high"] 

210 health_score = (success_rate * 0.4) + (quality_score * 0.4) + (0.2 if not critical_alerts else 0.0) 

211 return self.helpers.categorize_health_score(health_score)
