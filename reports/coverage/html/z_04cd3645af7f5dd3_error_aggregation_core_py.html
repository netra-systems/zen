23class ErrorAggregator: 

24 """Aggregates errors into patterns with time-based cleanup.""" 

25 

26 def __init__(self, window_size_minutes: int = 60, max_samples_per_pattern: int = 10): 

27 """Initialize error aggregator with configurable parameters.""" 

28 self.window_size = timedelta(minutes=window_size_minutes) 

29 self.max_samples = max_samples_per_pattern 

30 self.signature_extractor = ErrorSignatureExtractor() 

31 self.patterns: Dict[str, ErrorPattern] = {} 

32 self.error_history: List[tuple[datetime, Dict[str, Any]]] = [] 

33 self.last_cleanup = datetime.now() 

34 self.cleanup_interval = timedelta(hours=1) 

35 

36 def add_error(self, error_data: Dict[str, Any]) -> ErrorPattern: 

37 """Add error to aggregation pipeline.""" 

38 timestamp = self._extract_timestamp(error_data) 

39 signature = self.signature_extractor.extract_signature(error_data) 

40 pattern = self._get_or_create_pattern(signature) 

41 self._complete_error_processing(pattern, error_data, timestamp) 

42 return pattern 

43 

44 def _complete_error_processing(self, pattern: ErrorPattern, error_data: Dict[str, Any], timestamp: datetime) -> None: 

45 """Complete error processing pipeline.""" 

46 self._process_error_data(pattern, error_data, timestamp) 

47 self._add_to_history(timestamp, error_data) 

48 self._cleanup_if_needed() 

49 

50 def get_patterns_in_window(self, window_minutes: int = 60) -> List[ErrorPattern]: 

51 """Get error patterns within specified time window.""" 

52 cutoff_time = datetime.now() - timedelta(minutes=window_minutes) 

53 return [ 

54 pattern for pattern in self.patterns.values() 

55 if pattern.last_occurrence >= cutoff_time 

56 ] 

57 

58 def get_top_patterns(self, limit: int = 10, window_minutes: int = 60) -> List[ErrorPattern]: 

59 """Get top error patterns by count within window.""" 

60 patterns = self.get_patterns_in_window(window_minutes) 

61 return sorted(patterns, key=lambda p: p.count, reverse=True)[:limit] 

62 

63 def _extract_timestamp(self, error_data: Dict[str, Any]) -> datetime: 

64 """Extract and validate timestamp from error data.""" 

65 timestamp = error_data.get('timestamp', datetime.now()) 

66 if isinstance(timestamp, str): 

67 timestamp = datetime.fromisoformat(timestamp) 

68 return timestamp 

69 

70 def _process_error_data(self, pattern: ErrorPattern, error_data: Dict[str, Any], timestamp: datetime) -> None: 

71 """Process error data for pattern update.""" 

72 self._update_pattern(pattern, error_data, timestamp) 

73 self._store_sample_error(pattern, error_data, timestamp) 

74 

75 def _get_or_create_pattern(self, signature) -> ErrorPattern: 

76 """Get existing pattern or create new one.""" 

77 pattern = self.patterns.get(signature.pattern_hash) 

78 if not pattern: 

79 pattern = ErrorPattern(signature=signature) 

80 self.patterns[signature.pattern_hash] = pattern 

81 return pattern 

82 

83 def _update_pattern(self, pattern: ErrorPattern, error_data: Dict[str, Any], timestamp: datetime) -> None: 

84 """Update pattern with new error occurrence.""" 

85 pattern.count += 1 

86 pattern.last_occurrence = timestamp 

87 self._update_severity_distribution(pattern, error_data) 

88 self._track_affected_entities(pattern, error_data) 

89 

90 def _update_severity_distribution(self, pattern: ErrorPattern, error_data: Dict[str, Any]) -> None: 

91 """Update pattern severity distribution.""" 

92 severity = error_data.get('severity', ErrorSeverity.MEDIUM) 

93 if isinstance(severity, str): 

94 severity = ErrorSeverity(severity) 

95 pattern.severity_distribution[severity] += 1 

96 

97 def _track_affected_entities(self, pattern: ErrorPattern, error_data: Dict[str, Any]) -> None: 

98 """Track affected users and operations.""" 

99 if 'user_id' in error_data: 

100 pattern.affected_users.add(error_data['user_id']) 

101 if 'operation_id' in error_data: 

102 pattern.affected_operations.add(error_data['operation_id']) 

103 

104 def _store_sample_error(self, pattern: ErrorPattern, error_data: Dict[str, Any], timestamp: datetime) -> None: 

105 """Store sample error if under limit.""" 

106 if len(pattern.sample_errors) < self.max_samples: 

107 sample = self._create_sample_entry(error_data, timestamp) 

108 pattern.sample_errors.append(sample) 

109 

110 def _create_sample_entry(self, error_data: Dict[str, Any], timestamp: datetime) -> Dict[str, Any]: 

111 """Create sample error entry.""" 

112 return { 

113 'timestamp': timestamp.isoformat(), 

114 'message': error_data.get('message', ''), 

115 'context': error_data.get('context', {}) 

116 } 

117 

118 def _add_to_history(self, timestamp: datetime, error_data: Dict[str, Any]) -> None: 

119 """Add error to historical tracking.""" 

120 self.error_history.append((timestamp, error_data)) 

121 

122 def _cleanup_if_needed(self) -> None: 

123 """Trigger cleanup if interval elapsed.""" 

124 if datetime.now() - self.last_cleanup > self.cleanup_interval: 

125 self._cleanup_old_data() 

126 

127 def _cleanup_old_data(self) -> None: 

128 """Clean up old error data beyond retention period.""" 

129 cutoff_time = datetime.now() - timedelta(days=7) 

130 self._cleanup_history(cutoff_time) 

131 removed_count = self._cleanup_patterns(cutoff_time) 

132 self.last_cleanup = datetime.now() 

133 self._log_cleanup_results(removed_count) 

134 

135 def _cleanup_history(self, cutoff_time: datetime) -> None: 

136 """Remove old entries from error history.""" 

137 self.error_history = [ 

138 (timestamp, data) for timestamp, data in self.error_history 

139 if timestamp >= cutoff_time 

140 ] 

141 

142 def _cleanup_patterns(self, cutoff_time: datetime) -> int: 

143 """Remove old patterns and return count.""" 

144 patterns_to_remove = self._identify_old_patterns(cutoff_time) 

145 self._remove_old_patterns(patterns_to_remove) 

146 return len(patterns_to_remove) 

147 

148 def _identify_old_patterns(self, cutoff_time: datetime) -> List[str]: 

149 """Identify patterns older than cutoff time.""" 

150 return [ 

151 pattern_hash for pattern_hash, pattern in self.patterns.items() 

152 if pattern.last_occurrence < cutoff_time 

153 ] 

154 

155 def _remove_old_patterns(self, patterns_to_remove: List[str]) -> None: 

156 """Remove old patterns from storage.""" 

157 for pattern_hash in patterns_to_remove: 

158 del self.patterns[pattern_hash] 

159 

160 def _log_cleanup_results(self, removed_count: int) -> None: 

161 """Log cleanup operation results.""" 

162 logger.info( 

163 f"Cleaned up {removed_count} old patterns, " 

164 f"{len(self.error_history)} history entries remain" 

165 )

168class ErrorAggregationSystem: 

169 """Main error aggregation and alerting system orchestrator.""" 

170 

171 def __init__(self): 

172 """Initialize aggregation system components.""" 

173 self.aggregator = ErrorAggregator() 

174 self.trend_analyzer = ErrorTrendAnalyzer() 

175 self.alert_engine = AlertEngine() 

176 self.processing_active = False 

177 self.process_task: Optional[asyncio.Task] = None 

178 self.process_interval = 30 # seconds 

179 

180 async def process_error(self, error_data: Dict[str, Any]) -> None: 

181 """Process single error through complete pipeline.""" 

182 pattern = self.aggregator.add_error(error_data) 

183 trend = self._analyze_pattern_trend(pattern) 

184 alerts = self.alert_engine.evaluate_pattern(pattern, trend) 

185 self._log_generated_alerts(alerts) 

186 

187 async def start_processing(self) -> None: 

188 """Start background processing if not active.""" 

189 if self.processing_active: 

190 return 

191 self.processing_active = True 

192 self.process_task = asyncio.create_task(self._processing_loop()) 

193 logger.info("Error aggregation processing started") 

194 

195 async def stop_processing(self) -> None: 

196 """Stop background processing gracefully.""" 

197 self.processing_active = False 

198 if self.process_task: 

199 await self._cancel_process_task() 

200 logger.info("Error aggregation processing stopped") 

201 

202 def get_system_status(self) -> Dict[str, Any]: 

203 """Get comprehensive system status and metrics.""" 

204 basic_status = self._get_basic_status() 

205 extended_status = self._get_extended_status() 

206 return {**basic_status, **extended_status} 

207 

208 def _get_basic_status(self) -> Dict[str, Any]: 

209 """Get basic system status metrics.""" 

210 return { 

211 'total_patterns': len(self.aggregator.patterns), 

212 'active_patterns': len(self.aggregator.get_patterns_in_window(60)), 

213 'total_alerts': len(self.alert_engine.alerts), 

214 'processing_active': self.processing_active 

215 } 

216 

217 def _get_extended_status(self) -> Dict[str, Any]: 

218 """Get extended system status metrics.""" 

219 return { 

220 'unresolved_alerts': self._count_unresolved_alerts(), 

221 'top_patterns': self._get_top_patterns_summary() 

222 } 

223 

224 def _analyze_pattern_trend(self, pattern: ErrorPattern): 

225 """Analyze trend for given pattern.""" 

226 return self.trend_analyzer.analyze_pattern_trend( 

227 pattern, self.aggregator.error_history 

228 ) 

229 

230 def _log_generated_alerts(self, alerts: List) -> None: 

231 """Log newly generated alerts.""" 

232 for alert in alerts: 

233 logger.warning(f"Error alert generated: {alert.message}") 

234 

235 async def _cancel_process_task(self) -> None: 

236 """Cancel processing task with proper exception handling.""" 

237 self.process_task.cancel() 

238 try: 

239 await self.process_task 

240 except asyncio.CancelledError: 

241 pass 

242 

243 async def _processing_loop(self) -> None: 

244 """Background processing loop for periodic analysis.""" 

245 while self.processing_active: 

246 if not await self._try_process_patterns(): 

247 break 

248 

249 async def _try_process_patterns(self) -> bool: 

250 """Try to process patterns with error handling.""" 

251 try: 

252 await self._process_all_patterns() 

253 await asyncio.sleep(self.process_interval) 

254 return True 

255 except asyncio.CancelledError: 

256 return False 

257 except Exception as e: 

258 await self._handle_processing_error(e) 

259 return True 

260 

261 async def _handle_processing_error(self, error: Exception) -> None: 

262 """Handle processing loop errors.""" 

263 logger.error(f"Error aggregation processing error: {error}") 

264 await asyncio.sleep(self.process_interval) 

265 

266 async def _process_all_patterns(self) -> None: 

267 """Process all patterns for trend analysis and alerting.""" 

268 patterns = self.aggregator.get_patterns_in_window(60) 

269 for pattern in patterns: 

270 trend = self._analyze_pattern_trend(pattern) 

271 self.alert_engine.evaluate_pattern(pattern, trend) 

272 

273 def _count_unresolved_alerts(self) -> int: 

274 """Count unresolved alerts.""" 

275 return len([ 

276 a for a in self.alert_engine.alerts 

277 if not a.resolved 

278 ]) 

279 

280 def _get_top_patterns_summary(self) -> List[Dict[str, Any]]: 

281 """Get summary of top error patterns.""" 

282 top_patterns = self.aggregator.get_top_patterns(5) 

283 return [self._create_pattern_summary(p) for p in top_patterns]

285 def _create_pattern_summary(self, pattern: ErrorPattern) -> Dict[str, Any]: 

286 """Create summary for single pattern.""" 

287 return { 

288 'signature': pattern.signature.pattern_hash, 

289 'count': pattern.count, 

290 'error_type': pattern.signature.error_type, 

291 'module': pattern.signature.module
