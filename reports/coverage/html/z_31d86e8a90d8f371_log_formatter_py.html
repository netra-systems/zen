184def _check_priority_categories(prompt_lower: str) -> str: 

185 """Checks optimization and error categories first.""" 

186 if _contains_optimization_words(prompt_lower): 

187 return _get_optimization_response() 

188 elif _contains_error_words(prompt_lower): 

189 return _get_error_response()

242def _create_generic_response(user_prompt: str) -> str: 

243 """Creates a generic response for unmatched prompts.""" 

244 truncated = user_prompt[:50]

260def _update_messages_for_multi_turn(log: Dict[str, Any], row: Dict[str, Any]) -> None: 

261 """Updates log messages for multi-turn conversation format.""" 

262 context_response = _generate_contextual_response(row["user_prompt"]) 

263 log["request"]["prompt"]["messages"] = [
