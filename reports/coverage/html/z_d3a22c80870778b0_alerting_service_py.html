99 async def acknowledge_alert(self, alert_id: str, acknowledged_by: str) -> bool: 

100 """Acknowledge an alert""" 

101 if alert_id not in self.alerts: 

102 return False 

103 

104 alert = self.alerts[alert_id] 

105 if alert.status == AlertStatus.ACTIVE: 

106 alert.status = AlertStatus.ACKNOWLEDGED 

107 alert.acknowledged_at = datetime.now(UTC) 

108 alert.updated_at = alert.acknowledged_at 

109 alert.metadata["acknowledged_by"] = acknowledged_by

112 return True

116 """Resolve an alert""" 

117 if alert_id not in self.alerts: 

118 return False 

119 

120 alert = self.alerts[alert_id] 

121 if alert.status in [AlertStatus.ACTIVE, AlertStatus.ACKNOWLEDGED]:

177 async def _process_alert(self, alert: Alert) -> None: 

178 """Process alert through handlers and rules""" 

179 # Check suppression rules 

180 if await self._should_suppress_alert(alert): 

181 await self.suppress_alert(alert.alert_id, "Suppression rule matched") 

182 return 

183 

184 # Process through handlers 

185 for handler in self.alert_handlers: 

186 try: 

187 await handler(alert) 

188 except Exception as e: 

189 logger.error(f"Error in alert handler: {e}")

191 async def _should_suppress_alert(self, alert: Alert) -> bool: 

192 """Check if alert should be suppressed based on rules""" 

193 for rule in self.suppression_rules: 

194 if await self._matches_suppression_rule(alert, rule): 

195 return True 

196 return False

198 async def _matches_suppression_rule(self, alert: Alert, rule: Dict[str, Any]) -> bool: 

199 """Check if alert matches suppression rule""" 

200 # Simple rule matching - can be extended 

201 if "source" in rule and alert.source != rule["source"]: 

202 return False 

203 if "severity" in rule and alert.severity != AlertSeverity(rule["severity"]): 

204 return False 

205 if "pattern" in rule and rule["pattern"] not in alert.title: 

206 return False 

207 return True
